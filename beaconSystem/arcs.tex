\documentclass[tikz,border=10pt]{standalone}
\usepackage{verbatim}

\usetikzlibrary{calc}
\def\beaconsystem[#1][#2](#3:#4:#5:#6)% Syntax: [focus draw options][large angle draw options](number of nodes:radius:sensor error:focus angle)
{
  \foreach \s in {1,...,#3}
  {
   \node[draw,circle] (\s) at ({360/#3 * (\s - 1)}:#4) {$\s$};
   \draw[#1] ([shift=({360/#3 * (\s - 1)+180-#6}:#4+#5)]\s) arc ({360/#3 * (\s - 1)+180-#6}:{(360/#3*(\s - 1))+180+#6}:#4+#5);
   \draw[#1] ([shift=({360/#3 * (\s - 1)+180-#6}:#4-#5)]\s) arc ({360/#3 * (\s - 1)+180-#6}:{(360/#3*(\s - 1))+180+#6}:#4-#5);

   \draw[#2] ([shift=({360/#3 * (\s - 1)+180-#6*3}:#4+#5)]\s) arc ({360/#3 * (\s - 1)+180-#6*3}:{(360/#3*(\s - 1))+180+#6*3}:#4+#5);
   \draw[#2] ([shift=({360/#3 * (\s - 1)+180-#6*3}:#4-#5)]\s) arc ({360/#3 * (\s - 1)+180-#6*3}:{(360/#3*(\s - 1))+180+#6*3}:#4-#5);
  }
  \node (origin) at (0,0) [point, label = {below right:$Drone$}]{};
}
\begin{document}
\begin{tikzpicture}
 [
  scale=3,
  >=stealth,
  point/.style = {draw, circle,  fill = black, inner sep = 1pt},
  dot/.style   = {draw, circle,  fill = black, inner sep = .2pt},
 ]

 \beaconsystem[thin,red][dashed,red](50:2:0.1:15)% Syntax: [focus draw options][large angle draw options](number of nodes:radius:sensor error:focus angle)

\end{tikzpicture}
\end{document}
